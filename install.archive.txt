function install_mapdrives {
# Setup Network Mounts
	echo "*** Setting common network mount points"
	sudo apt-get install smbfs

	echo -e "username=lorne\npassword=lornin" | sudo tee /root/creds
	sudo chmod 400 /root/creds

	function map {
		if [[ -z `grep "$1" /etc/fstab` ]]; then
			[ -d $2  ] || sudo mkdir $2
			echo "$1	$2 cifs credentials=/root/creds,uid=lorne,_netdev 0 0" | sudo tee -a /etc/fstab
		fi
	}

	map //server/backup		/media/backup
	map //server/documents	/media/documents
	map //server/repos		/media/repos
	map //server/www		/media/www
	map //desktop/storage	/media/desktop
	map //desktop/data		/media/data
	map //desktop/media		/media/media
}

function install_xmonad {
	echo "*** Installing xmonad..."
	sudo apt-get install xmonad
	sudo apt-get install libghc6-xmonad-dev
	sudo apt-get install libghc6-xmonad-contrib-dev
	[ -d ~/.xmonad ] || mkdir ~/.xmonad
	dpkg --get-selections | grep 'gnome-do\s*install'
	if [[ `dpkg --get-selections | grep 'gnome-do\s*install'` ]]; then
		cp -v apps/xmonad/xmonad.gnomedo.hs ~/.xmonad/xmonad.hs
	else
		cp -v apps/xmonad/xmonad.gnome.hs ~/.xmonad/xmonad.hs
	fi
	cp -v scripts/wmswap.sh ~/scripts
}

function install_yubikey {
	echo "*** Installing yubikey support..."
	yubipam="YubiPAM-1.0.4"
	cd $yubipam
	sudo apt-get install libpam0g-dev
	sudo make clean
	sudo make distclean
	sudo ./configure
	sudo make
	sudo make install

	sudo addgroup yubiauth
	sudo touch /etc/yubikey
	sudo chgrp yubiauth /etc/yubikey /sbin/yk_chkpwd
	sudo chmod g+rw /etc/yubikey
	sudo chmod g+s /sbin/yk_chkpwd

	if [[ -z `grep "auth sufficient pam_yubikey.so" /etc/pam.d/common-auth` ]]; then
		echo "auth sufficient pam_yubikey.so" > temp
		cat /etc/pam.d/common-auth >> temp
		sudo sed -i.bak 's/nullok_secure$/nullok_secure try_first_pass/' temp
		sudo cp /etc/pam.d/common-auth /etc/pam.d/common-auth.bak
		sudo mv temp /etc/pam.d/common-auth
	fi
	cd ..
}

function sub_boxes {
	echo "*** Installing boxes..."
	cd apps/boxes
	make
	sudo cp -v src/boxes /usr/bin
	sudo cp -v boxes ~/.boxes
	sudo cp -v boxes-config /usr/share/boxes
	sudo cp -v doc/boxes.1 /usr/share/man/man1
	sudo chmod 755 /usr/bin/boxes /usr/share/boxes /usr/share/man/man1/boxes.1
	cd $home
}
